n=int(input())
Han=[i for i in range(1,100)]
for a in range(1,10):
  for d in range(0,a//2+1):
    Han.append(a*100+(a-d)*10+a-2*d)
    if d!=0 and a-2*d>0: Han.append((a-2*d)*100+(a-d)*10+a)
if n<100: print(n)
elif n<111: print(99)
else:
  Han.sort()
  for i in range(143,98,-1): #Han[98]=99
    if Han[i]<=n: break
  print(i+1)
